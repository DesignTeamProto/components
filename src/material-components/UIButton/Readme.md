### Import component

```jsx static
import UIButton from '../UIButton';
```

### Text buttons

```jsx
<UIButton>Default</UIButton>
<UIButton color="primary">Primary</UIButton>
<UIButton color="secondary">Secondary</UIButton>
<UIButton disabled>Disabled</UIButton>
<UIButton href="#button">Link</UIButton>
```

### Contained buttons

```jsx
<UIButton variant="contained">Default</UIButton>
<UIButton variant="contained" color="primary">Primary</UIButton>
<UIButton variant="contained" color="secondary">Secondary</UIButton>
<UIButton variant="contained" disabled>Disabled</UIButton>
<UIButton variant="contained" href="#button">Link</UIButton>
```

### Outlined buttons

```jsx
<UIButton variant="outlined">Default</UIButton>
<UIButton variant="outlined" color="primary">Primary</UIButton>
<UIButton variant="outlined" color="secondary">Secondary</UIButton>
<UIButton variant="outlined" disabled>Disabled</UIButton>
<UIButton variant="outlined" href="#button">Link</UIButton>
```

### Floating action buttons

```jsx
<UIButton variant="fab" color="primary" aria-label="Search">A</UIButton>
<UIButton variant="fab" color="secondary" aria-label="Search">B</UIButton>
<UIButton variant="extendedFab" aria-label="Directions">Directions</UIButton>
<UIButton variant="fab" disabled aria-label="Delete">D</UIButton>
```

### Sizes

```jsx
<UIButton variant="contained" color="primary" size="small">Small</UIButton>
<UIButton variant="contained" color="primary" size="medium">Medium</UIButton>
<UIButton variant="contained" color="primary" size="large">Large</UIButton>
```
